CLASS lsc_yas_i_travel_root DEFINITION INHERITING FROM cl_abap_behavior_saver.

  PROTECTED SECTION.

    METHODS save_modified REDEFINITION.

ENDCLASS.

CLASS lsc_yas_i_travel_root IMPLEMENTATION.

  METHOD save_modified.

    DATA:
      lt_log   TYPE STANDARD TABLE OF yas_t_travel_log,
      lt_log_c TYPE STANDARD TABLE OF yas_t_travel_log,
      lt_log_u TYPE STANDARD TABLE OF yas_t_travel_log.

    IF create-travel IS NOT INITIAL.

      lt_log = CORRESPONDING #( create-travel ).

      LOOP AT lt_log ASSIGNING FIELD-SYMBOL(<ls_log>).

        <ls_log>-changing_operation = 'CREATE'.
        GET TIME STAMP FIELD <ls_log>-created_at.

        READ TABLE create-travel WITH TABLE KEY entity
                                                COMPONENTS
                                                TravelId = <ls_log>-travelid
                                                ASSIGNING FIELD-SYMBOL(<ls_travel>).
        IF sy-subrc IS INITIAL.

          " Populate the log information for Travel with field Booking Fee.
          IF <ls_travel>-%control-BookingFee = cl_abap_behv=>flag_changed.

            <ls_log>-changed_field_name = 'Booking Fee'.
            <ls_log>-changed_value = <ls_travel>-BookingFee.

            TRY.
                <ls_log>-change_id = cl_system_uuid=>create_uuid_x16_static( ).

              CATCH cx_uuid_error.

            ENDTRY.

            APPEND <ls_log> TO lt_log_c.

          ENDIF.

          " Populate the log information for Travel with field Overall Status.
          IF <ls_travel>-%control-OverallStatus = cl_abap_behv=>flag_changed.

            <ls_log>-changed_field_name = 'Overall Status'.
            <ls_log>-changed_value      = <ls_travel>-OverallStatus.

            TRY.
                <ls_log>-change_id = cl_system_uuid=>create_uuid_x16_static( ).

              CATCH cx_uuid_error.

            ENDTRY.

            APPEND <ls_log> TO lt_log_c.

          ENDIF.

        ENDIF.

      ENDLOOP.

      INSERT yas_t_travel_log FROM TABLE @lt_log_c.

    ENDIF.

    IF update-travel IS NOT INITIAL.

      IF <ls_log> IS ASSIGNED.
        UNASSIGN <ls_log>.
      ENDIF.

      IF <ls_travel> IS ASSIGNED.
        UNASSIGN <ls_travel>.
      ENDIF.

      lt_log = CORRESPONDING #( update-travel ).

      LOOP AT lt_log ASSIGNING <ls_log>.

        <ls_log>-changing_operation = 'UPDATE'.
        GET TIME STAMP FIELD <ls_log>-created_at.

        READ TABLE update-travel WITH TABLE KEY entity
                                                COMPONENTS
                                                TravelId = <ls_log>-travelid
                                                ASSIGNING <ls_travel>.
        IF sy-subrc IS INITIAL.

          " Populate the log information for Travel with field customer ID.
          IF <ls_travel>-%control-CustomerId = cl_abap_behv=>flag_changed.

            <ls_log>-changed_field_name = 'Customer Id'.
            <ls_log>-changed_value = <ls_travel>-CustomerId.

            TRY.
                <ls_log>-change_id = cl_system_uuid=>create_uuid_x16_static( ).

              CATCH cx_uuid_error.

            ENDTRY.

            APPEND <ls_log> TO lt_log_u.

          ENDIF.

          " Populate the log information for Travel with field Overall Status.
          IF <ls_travel>-%control-Description = cl_abap_behv=>flag_changed.

            <ls_log>-changed_field_name = 'Description'.
            <ls_log>-changed_value      = <ls_travel>-Description.

            TRY.
                <ls_log>-change_id = cl_system_uuid=>create_uuid_x16_static( ).

              CATCH cx_uuid_error.

            ENDTRY.

            APPEND <ls_log> TO lt_log_u.

          ENDIF.

        ENDIF.

      ENDLOOP.

      INSERT yas_t_travel_log FROM TABLE @lt_log_u.

    ENDIF.

    IF delete-travel IS NOT INITIAL.

      IF <ls_log> IS ASSIGNED.
        UNASSIGN <ls_log>.
      ENDIF.

      IF <ls_travel> IS ASSIGNED.
        UNASSIGN <ls_travel>.
      ENDIF.

      lt_log = CORRESPONDING #( delete-travel ).

      LOOP AT lt_log ASSIGNING <ls_log>.

        <ls_log>-changing_operation = 'DELETE'.
        GET TIME STAMP FIELD <ls_log>-created_at.

        TRY.
            <ls_log>-change_id = cl_system_uuid=>create_uuid_x16_static( ).

          CATCH cx_uuid_error.

        ENDTRY.

        APPEND <ls_log> TO lt_log_u.

      ENDLOOP.

      INSERT yas_t_travel_log FROM TABLE @lt_log_u.

    ENDIF.

  ENDMETHOD.

ENDCLASS.
